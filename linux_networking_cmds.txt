1.ping <ip address or hostname> - to check connection
2.ssh <hostname to connect> - to connect remotely to other system
3.ip address - list details
4.ip a - show devices ip address
5.ip route - show routes / rules
6.dig <ip address> - to see info or details
7.iperf3 -s - tool used to test network speed
8. traceroute <hostname> - to see the path (Hop by Hop)
9.arp -a - to see ip to mac mapping
10.nslookup <hostname> - to see DNS
11.ss -tuln (netstat -tuln) - to see active connection
12.cat /var/lib/dhcp/dhclient.leases - to check DHCP lease
13.sudo tcpdump - packet Sniffing
14.sudo ufw status - to check firewall status
15.sudo ufw deny icmp - to deny or block ping
16.sudo ufw allow icmp - to allow
17.curl google.com - Test web
18.ip link show - to see interface(physical/datalink layer)
19.sudo ip link set wlan0 down - turn off network
20.sudo ip link set wlan0 up - turn on
